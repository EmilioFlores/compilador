




funcion entero fibonacciRecursivo(entero n){

		var entero ramaIzquierda;
		var entero ramaDerecha;
		condicion(n == 0){
			n = 0;

		} fallo {
			condicion(n < 3){
				n = 1;
			} fallo {

				
				ramaIzquierda = fibonacciRecursivo(n - 1);
				ramaDerecha = fibonacciRecursivo(n - 2);
				n = ramaIzquierda + ramaDerecha ;
			};
		};
		regresa n;
	};
