




funcion entero fibIterativo(entero n)
{
    var entero a ;
    var entero b ;
    var entero c ;
    var entero i ;
    a = 1;
    b = 1;


    ciclo ( i = 3, i <= n, i = i + 1) {
    	

        c = a + b;
        a = b;
        b = c;
    };          
	
    regresa b;
};
